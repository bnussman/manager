export const Template = (args) => {
  const heads = [];
  const unit = 100 / args.colSpan;
  for (let i = 0; i < args.colSpan; i++) {
    heads.push(unit);
  }
  return (
    <Table>
      {renderTableHead(...heads)}
      <TableBody>
        <TableRowLoading {...args} />
      </TableBody>
    </Table>
  );
};
